# floodings
########################################################################################
country_event = {
	id = natural_disaster.1
	title = "natural_disaster.1.t"
	desc = "natural_disaster.1.d"
	picture = flooding
	is_triggered_only = yes
	goto = flooded_provinces
	
	trigger = {
		any_owned_province = {
			development = 2
			NOT = { has_province_modifier = floodings_modifier_1 }
			NOT = { has_province_modifier = floodings_modifier_2 }
			OR = {
				has_terrain = mesopotamian_plain
				has_terrain = babil_t
				has_terrain = uruk_t
				has_terrain = nile_plain
				has_terrain = memphis_t
				has_terrain = waset_t
				has_terrain = elephantine_t
				has_terrain = floodplains
				has_terrain = alluvial_plain
				has_terrain = marsh
				has_terrain = mangrove
				has_terrain = wetlands
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					development = 2
					NOT = { has_province_modifier = floodings_modifier_1 }
					NOT = { has_province_modifier = floodings_modifier_2 }
					OR = {
						has_terrain = mesopotamian_plain
						has_terrain = babil_t
						has_terrain = uruk_t
						has_terrain = nile_plain
						has_terrain = memphis_t
						has_terrain = waset_t
						has_terrain = elephantine_t
						has_terrain = floodplains
						has_terrain = alluvial_plain
						has_terrain = marsh
						has_terrain = mangrove
						has_terrain = wetlands
					}
				}
				save_event_target_as = flooded_provinces
			}
		}
	}

	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 2
			event_target:flooded_provinces = {
				OR = {
					region = mesopotamia_region
					region = lower_egypt_region
					region = upper_egypt_region
					region = sindhu_region
					region = vanga_region
				}
			}
		}
	}
	
	option = {
		name = "natural_disaster.1.a"
		ai_chance = {
			factor = 40
			modifier = {
				factor = 2
				NOT = { years_of_income = 0.2 }
			}
			modifier = {
				factor = 0
				is_bankrupt = yes
			}
		}
		add_prestige = -4
		event_target:flooded_provinces = {
			area = {
				limit = {
					OR = {
						has_terrain = mesopotamian_plain
						has_terrain = babil_t
						has_terrain = uruk_t
						has_terrain = nile_plain
						has_terrain = memphis_t
						has_terrain = waset_t
						has_terrain = elephantine_t
						has_terrain = floodplains
						has_terrain = alluvial_plain
						has_terrain = marsh
						has_terrain = mangrove
						has_terrain = wetlands
					}
				}
				add_permanent_province_modifier = {
					name = floodings_modifier_1
					duration = 1825
				}
				if = {
					limit = { base_manpower = 3 }
					add_base_manpower = -2
					change_variable = { PS_rur_evtd = 2 }
					add_devastation = 60
				}
				else_if = {
					limit = { base_manpower = 2 }
					add_base_manpower = -1
					change_variable = { PS_rur_evtd = 1 }
					add_devastation = 60
				}
				else_if = {
					limit = { base_manpower = 1 }
					add_devastation = 60
				}
			}
		}
	}
	option = {
		name = "natural_disaster.1.b"
		ai_chance = {
			factor = 60
			modifier = {
				factor = 3
				years_of_income = 0.6
			}
		}
		add_years_of_income = -0.2
		add_prestige = -1
		event_target:flooded_provinces = {
			area = {
				limit = {
					OR = {
						has_terrain = mesopotamian_plain
						has_terrain = babil_t
						has_terrain = uruk_t
						has_terrain = nile_plain
						has_terrain = memphis_t
						has_terrain = waset_t
						has_terrain = elephantine_t
						has_terrain = floodplains
						has_terrain = alluvial_plain
						has_terrain = marsh
						has_terrain = mangrove
						has_terrain = wetlands
					}
				}
				add_devastation = 50
				add_permanent_province_modifier = {
					name = floodings_modifier_2
					duration = 1095
				}
				if = {
					limit = { base_manpower = 2 }
					add_base_manpower = -1
