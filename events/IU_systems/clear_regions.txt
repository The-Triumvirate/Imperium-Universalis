# Clear britannia_superregion
country_event = {
	id = clear_regions.201
	title = "clear_regions.201.t"
	desc = "clear_regions.201.d"
	picture = fingersnap_marduk

	is_triggered_only = yes

	option = {
		name = "clear_regions.201.a"
		hidden_effect = {
			colonizable_disable_superregion = { superregion = britannia_superregion }
		}
	}

	option = {
		name = "clear_regions.201.b"
		hidden_effect = {
			complete_disable_superregion = { superregion = britannia_superregion }
		}
	}

	option = {
		name = "clear_regions.201.c"
		custom_tooltip = COUNTRY_TARGETING_DISABLING_TOOLTIP
		hidden_effect = {
			country_disable_superregion = { superregion = britannia_superregion }
		}
	}

	option = { # Go Back
		name = "clear_regions.1.n"
	}

	after = {
		if = {
			limit = { is_europe_inactive = no }
			country_event = { id = clear_regions.2 }
		}
		else = {
			country_event = { id = clear_regions.1 }
		}
	}
}

	id = clear_regions.6
	title = "clear_regions.6.t"
	desc = "clear_regions.2.d"
	picture = fingersnap_marduk

	is_triggered_only = yes

	option = { # Clear Uttarapatha
		name = "clear_regions.6.a"
		trigger = {
			NOT = { has_global_flag = picked_disable_uttarapatha_superregion_flag }
		}
		country_event = {
			id = clear_regions.601
		}
	}

	option = { # Clear Dakshinapatha
		name = "clear_regions.6.b"
		trigger = {
			NOT = { has_global_flag = picked_disable_dakshinapatha_superregion_flag }
		}
		country_event = {
			id = clear_regions.602
		}
	}

	option = { # Clear Tibet
		name = "clear_regions.6.c"
		trigger = {
			NOT = { has_global_flag = picked_disable_tibet_superregion_flag }
		}
		country_event = {
			id = clear_regions.603
		}
	}

	option = { # Clear Suvarnabhumi
		name = "clear_regions.6.e"
		trigger = {
			NOT = { has_global_flag = picked_disable_suvarnabhumi_superregion_flag }
		}
		country_event = {
			id = clear_regions.604
		}
	}

	option = { # Clear entire category
		name = "clear_regions.6.f"
		country_event = {
			id = clear_regions.605
		}
	}

	option = { # Go Back
		name = "clear_regions.1.n"
		country_event = {
			id = clear_regions.1
		}
	}

	option = { # No effect
		name = "clear_regions.1.m"
	}
}

# Clear uttarapatha_superregion
country_event = {
	id = clear_regions.601
	title = "clear_regions.601.t"
	desc = "clear_regions.201.d"
	picture = fingersnap_marduk

	is_triggered_only = yes

	option = {
		name = "clear_regions.201.a"
		hidden_effect = {
			colonizable_disable_superregion = { superregion = uttarapatha_superregion }
		}
	}

	option = {
		name = "clear_regions.201.b"
		hidden_effect = {
			complete_disable_superregion = { superregion = uttarapatha_superregion }
		}
	}

	option = {
		name = "clear_regions.201.c"
		custom_tooltip = COUNTRY_TARGETING_DISABLING_TOOLTIP
		hidden_effect = {
			country_disable_superregion = { superregion = uttarapatha_superregion }
		}
	}

	option = { # Go Back
		name = "clear_regions.1.n"
	}

	after = {
		if = {
			limit = { is_southern_asia_inactive = no }
			country_event = { id = clear_regions.6 }
		}
		else = {
			country_event = { id = clear_regions.1 }
		}
	}
}
