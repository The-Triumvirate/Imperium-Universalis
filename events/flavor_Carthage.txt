	id = carthage.9
	title = "carthage.9.t"
	desc = "carthage.9.d"

		OR = {
			primary_culture = phoenician
			primary_culture = punic
			primary_culture = ibero_phoenician
			primary_culture = sardo_phoenician
		}
		num_of_ports = 1
		any_country = {
			is_tribal = yes
			NOT = { culture_group = semitic }
			NOT = { border_distance = { who = ROOT  distance = 600 } }
			any_owned_province = {
				has_port = yes
				NOT = { has_province_modifier = phoenician_trade_outpost }
				OR = {
					superregion = italia_superregion
					superregion = western_africa_superregion
					superregion = celtica_superregion
					superregion = macedonia_superregion
					superregion = asia_superregion
					region = thracia_region
					region = sarmatia_region
					region = gallia_narbonensis
				}
			}
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_tribal = yes
				NOT = { culture_group = semitic }
				NOT = { border_distance = { who = ROOT  distance = 600 } }
				any_owned_province = {
					has_port = yes
					NOT = { has_province_modifier = phoenician_trade_outpost }
					OR = {
						superregion = italia_superregion
						superregion = western_africa_superregion
						superregion = celtica_superregion
						superregion = macedonia_superregion
						superregion = asia_superregion
						region = thracia_region
						region = sarmatia_region
						region = gallia_narbonensis
					}
				}
			}
			save_event_target_as = trade_parter_target
		}
	}

	option = {
		name = "greek.12.a"
		ai_chance = { factor = 90 }
		add_years_of_income = -0.1
		custom_tooltip = greek.12.tt
		hidden_effect = {
			event_target:trade_parter_target = {
				country_event = {
					id = carthage.10
					days = 90
				}
			}
		}
	}
	option = {
		name = "greek.12.b"
		highlight = yes
		trigger = {
			ruler_has_personality = entrepreneur_personality
		}
		ai_chance = { factor = 100 }
		add_dip_power = -25
		custom_tooltip = greek.12.tt
		hidden_effect = {
			event_target:trade_parter_target = {
				country_event = {
					id = carthage.10
					days = 90
				}
			}
		}
	}
	option = {
		name = "greek.12.c"

country_event = {
	id = carthage.11
	title = "carthage.11.t"
	desc = "carthage.11.d"
	picture = senat_conflict
	fire_only_once = yes

	is_triggered_only = yes

	trigger = {
		tag = CAR
		NOT = { has_reform = carthaginian_republic_reform }
		OR = {
			government = monarchy
			government = theocracy
		}
		owns_core_province = 612
		NOT = { mil = 4 }
		NOT = { absolutism = 30 }
	}

	option = {
		name = "carthage.11.a"
		ai_chance = { factor = 99 }
		add_stability = 1
		add_absolutism = 5
		add_dip_power = -50
		add_adm_power = -50
		if = {
			limit = { NOT = { government = theocracy } }
			change_government = theocracy
		}
		add_government_reform = carthaginian_republic_reform
	}
	option = {
		name = "carthage.11.b"
		ai_chance = { factor = 1 }
		add_stability = -1
		add_mil_power = 300
		capital_scope = {
			spawn_rebels = {
				size = 3
				type = republican_rebels
			}
		}
	}
}
