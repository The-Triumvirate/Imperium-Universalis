# Powerful Nomarch
########################################################################################
country_event = {
	id = new_governments.1
	title = "new_governments.1.t"
	desc = "new_governments.1.d"
	picture = battle

	is_triggered_only = yes

	trigger = {
		has_reform = kemetic_kingdom
		NOT = { absolutism = 25 }
		any_owned_province = {
			NOT = { has_province_modifier = powerful_local_nomarch }
			local_autonomy = 25
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_area = {
				limit = { is_state = yes }
				save_event_target_as = powerful_nomarch_area
			}
		}
	}

	option = {
		name = "new_governments.1.a"
		ai_chance = { factor = 60 }
		goto = powerful_nomarch_area
		event_target:powerful_nomarch_area = {
			add_province_modifier = {
				name = powerful_local_nomarch
				duration = 1825
			}
			add_local_autonomy = 10
		}
		add_faction_influence = {
			faction = fac_nomarchs
			influence = 5
		}
	}

	option = {
		name = "new_governments.1.b"
		ai_chance = { factor = 40 }
		add_faction_influence = {
			faction = fac_nomarchs
			influence = -5
		}
		add_absolutism = -5
	}
}

		hidden_effect = {

			}
		}

	option = {
		name = "new_governments.5.a"
		goto = province_priests_want_temple
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				treasury = 100
			}
		}
		event_target:province_priests_want_temple = {

			}
		}

			factor = 60

				factor = 0
				NOT = { personality = ai_militarist }

				factor = 2
				army_size_percentage = 0.6
			}
			modifier = {
				factor = 2
				war_exhaustion = 10
			}
		}

	}

				}

		}
	}
