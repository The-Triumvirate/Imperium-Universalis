country_event = {
	id = indian.1
	title = "indian.1.t"
	desc = "indian.1.d"
	picture = tradecity
	fire_only_once = yes
	major = yes

	trigger = {
		is_year = 254
		capital_scope = {
			region = northern_india_region
		}
		OR = {
			culture_group = indian_group
			primary_culture = asura
		}
		NOT = { started_in = 254.1.1 }
		OR = {
			has_reform = vedic_kingdom_reform
			have_had_reform = vedic_kingdom_reform
		}
		government_rank = 3
	}

	mean_time_to_happen = {
		months = 12
	}

	immediate = {
		hidden_effect = {
			set_global_flag = second_indian_urbanization
			set_country_flag = second_urbanization_janapada
		}
	}

	option = {
		name = "indian.1.a"
		ai_chance = { factor = 30 }
		every_owned_province = {
			if = {
				limit = {
					continent = india
					OR = {
						culture_group = indian_group
						culture = asura
					}
					base_manpower = 4
				}
				add_base_manpower = -3
				add_base_production = 3
				change_variable = { PS_rur_evtd = 3 }	change_variable = { PS_urb_evtu = 3 }
			}
		}
		add_country_modifier = {
			name = india_urban_explosion
			duration = 10950
		}
	}
}
