#When converting culture

province_event = {
	id = syncretized_culture.1
	title = syncretized_culture.1.t
	desc = syncretized_culture.1.d
	picture = forum_picture
	hidden = yes

	trigger = {
		owner = {
			OR = {
				primary_culture = latin
				primary_culture = gallo_roman
				primary_culture = hispano_roman
				primary_culture = afro_roman
				primary_culture = illyro_roman
				primary_culture = romano_pannonian
				primary_culture = romano_british
				primary_culture = thraco_roman
				primary_culture = daco_roman
				primary_culture = romano_noric
			}
		}
		has_construction = culture
		construction_progress = 0.1
		NOT = {
			OR = {
				region = gallia_cisalpina
				region = italian_region
				region = sardinia_corsica_region
				region = magna_graecia
			}
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		if = {
			limit = {
				OR = {
					culture_group = gaulish_group
					culture_group = north_atlantic_group
					culture_group = ligures_group
				}
			}
			set_province_flag = gallo_roman_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture_group = iberian_group
					culture_group = ibero_celtic_group
					culture_group = turdetani_group
					culture_group = aquitani_group
					culture_group = lusitanian_group
					culture = ibero_phoenician
					culture = sardo_phoenician
					culture = talaiotic
				}
			}
			set_province_flag = hispano_roman_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture_group = north_african_group
					culture = punic
					culture = lybo_punic
				}
			}
			set_province_flag = afro_roman_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture = iapyges
					culture = illyrian
					culture = ardiaei
					culture = dardani
					culture = liburni
				}
			}
			set_province_flag = illyro_roman_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture = pannonian
				}
			}
			set_province_flag = romano_pannonian_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture_group = britannic_group
				}
			}
			set_province_flag = romano_british_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture = odrysian
					culture = paeonian
				}
			}
			set_province_flag = thraco_roman_syncretizing_flag
		}
		if = {
			limit = {
				OR = {
					culture = dacian
					culture = getean
				}
			}
			set_province_flag = daco_roman_syncretizing_flag
		}
		if = {
			limit = { culture = raeti }
			set_province_flag = romano_noric_syncretizing_fla
		}
		if = {
			limit = {
				region = raetia_et_noricum
				culture_group = eastern_celtic
			}
			set_province_flag = romano_noric_syncretizing_flag
		}
		else_if = {
			limit = { culture_group = eastern_celtic }
			set_province_flag = gallo_roman_syncretizing_flag
		}
	}

	option = {
		name = syncretized_culture.1.a

	}
}
