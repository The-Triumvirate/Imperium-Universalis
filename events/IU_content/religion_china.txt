country_event = {
	id = religion_china.1
	title = "religion_china.1.t"
	desc = "religion_china.1.d"
	picture = ruler
	major = yes

	trigger = {
		culture_group = chinese_group
		religion_group = chinese_religions
		NOT = { is_religion_enabled = confucianism }
		OR = {
			reform_desire = 0.5
			AND = {
				religion = liyuebadao
				NOT = {
					any_known_country = {
						religion_group = chinese_religions
					}
				}
				has_unlocked_cult = xuanwu_worship
				has_unlocked_cult = qilin_worship
			}
			is_year = 254
		}
	}

	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 10
			current_age = age_of_empires
		}
		modifier = {
			factor = 5
			current_age = archaic_age
		}
		modifier = {
			factor = 0.75
			tag = LUU
		}
		modifier = {
			factor = 0.3
			has_global_flag = zhanguokaishi
		}
		modifier = {
			factor = 0.9
			NOT = {
				calc_true_if = {
					all_known_country = {
						religion = chinese_religion
						capital_scope = {
							continent = china
						}
					}
					amount = 35
				}
			}
		}
		modifier = {
			factor = 0.7
			NOT = {
				calc_true_if = {
					all_known_country = {
						religion = chinese_religion
						capital_scope = {
							continent = china
						}
					}
					amount = 30
				}
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				calc_true_if = {
					all_known_country = {
						religion = chinese_religion
						capital_scope = {
							continent = china
						}
					}
					amount = 25
				}
			}
		}
	}

	option = {
		name = "religion_china.1.a"
		define_advisor = {
			type = scholar_advisor
			name = "Confucius"
			skill = 3
			discount = yes
		}
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		enable_religion = confucianism
		add_reform_desire = -0.5
	}
}

########################################################################################
province_event = {
	id = religion_china.5
	title = "religion_china.5.t"
	desc = "religion_china.5.d"
	picture = religious

	trigger = {
		religion_group = eastern_religions
		any_neighbor_province = { religion_group = chinese_religions }
		owner = { religion_group = chinese_religions }
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.66
			any_neighbor_province = { religion_group = chinese_religions }
		}
		modifier = {
			factor = 0.75
			owner = { religion_group = chinese_religions }
		}
		modifier = {
			factor = 3.0
			owner = { NOT = { culture_group = chinese_group } }
			NOT = { culture_group = chinese_group }
		}
		modifier = {
			factor = 1.25
			owner = { NOT = { adm_tech = 4 } }
		}
		modifier = {
			factor = 0.8
			owner = { adm_tech = 9 }
		}
	}

	option = {
		name = "religion_china.5.a"
		change_religion = owner
	}
}
