			}

		}

		}
	}

		name = "ShiJian.9074.b"

			}

		}
		ROOT = {
			every_owned_province = {
				limit = {
					is_capital = no
				}
				cede_province = emperor
				add_core = emperor
			}
		}
		emperor = {
			vassalize = ROOT
		}
	}

	option = {
		name = "ShiJian.9074.c"
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				will_back_next_reform = yes
			}

		}
		every_owned_province = {
			limit = {
				is_part_of_hre = yes
			}
			add_permanent_claim = emperor
		}
		every_owned_province = {
			set_in_empire = no
		}
		emperor = {
			add_opinion = { who = ROOT modifier = opinion_left_empire }
		}
		add_opinion = { who = emperor modifier = opinion_united_empire }
		elector = no

	}
}
