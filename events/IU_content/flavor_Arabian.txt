### Shelter Jews
country_event = {
	id = arabia.4
	title = "arabia.4.t"
	desc = "arabia.4.d"
	picture = seaside_prayer

	trigger = {
		religion = arab_religion
		capital_scope = {
			superregion = arabia_superregion
		}
		OR = {
			AND = { #Collapse of Judah
				NOT = { is_year = 188 }
				is_year = 168
				NOT = {
					2277 = {
						owner = { religion = jewish }
					}
				}
			}
			AND = { #Roman Conquest of Judah
				is_year = 691
				NOT = {
					2277 = {
						owner = { religion = jewish }
					}
				}
				NOT = { is_year = 711 }
			}
			AND = { #Destruction of Jerusalem
				is_year = 823
				NOT = {
					2277 = {
						owner = { religion = jewish }
					}
				}
				NOT = { is_year = 843 }
			}
			AND = { #Bar Kochba revolt
				is_year = 888
				NOT = {
					2277 = {
						owner = { religion = jewish }
					}
				}
				NOT = { is_year = 908 }
			}
		}
		NOT = { has_country_modifier = jew_shelter_cm }
		NOT = { has_country_flag = jew_shelter_no_flag }
	}

	mean_time_to_happen = {
		months = 30
	}

	option = { #Provide
		name = "arabia.4.a"
		add_prestige = 10
		add_country_modifier = {
			name = jew_shelter_cm
			duration = 7300
		}

		random_owned_province = {
			limit = {
				is_capital = no
				NOT = { religion = jewish }
			}
			change_religion = jewish
			add_permanent_province_modifier = {
				name = jew_shelter_pm
				duration = -1
			}
		}
	}

	option = { #Decline
		name = "arabia.4.b"
		set_country_flag = jew_shelter_no_flag
	}
}
