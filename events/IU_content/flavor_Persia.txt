# Median-Persian War 1/3    - MDI (persia is no vassal)
########################################################################################
country_event = {
	id = persia.10
	title = "persia.10.t"
	desc = "persia.10.d"
	picture = rivercrossing
	fire_only_once = yes

	trigger = {
		tag = MDI
		NOT = { has_country_flag = attack_on_pse_flag }
		total_development = 65
		is_at_war = no
		NOT = { start_date = 195.1.1 }
		NOT = {  is_year = 225  }
		OR = {
			AND = {
				NOT = { exists = ANH }
				NOT = { exists = PRS }
				is_neighbor_of = PSE
				PSE = {
					is_subject = no
					num_of_cities = 8
				}
			}
			AND = {
				NOT = { exists = PSE }
				is_neighbor_of = ANH
				ANH = {
					is_subject = no
					num_of_cities = 14
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 72
		modifier = {
			factor = 0.5
			PSE = {
				ADM = 5
				DIP = 5
				MIL = 5
			}
		}
		modifier = {
			factor = 0.5
			ANH = {
				ADM = 5
				DIP = 5
				MIL = 5
			}
		}
		modifier = {
			factor = 0.5
			is_year = 205
		}
		modifier = {
			factor = 1.5
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1.5
			war_exhaustion = 2
		}
	}

	# Attack Persia
	option = {
		name = "persia.10.a"
		ai_chance = { factor = 99 }
		set_country_flag = attack_on_pse_flag
		custom_tooltip = cyrus_tooltip
		add_prestige = 5
		if = {
			limit = {
				exists = PSE
			}
			add_casus_belli = {
				target = PSE
				type = cb_disloyal_vassal
				months = 120
			}
			if = {
				limit = { ai = yes }
				declare_war_with_cb = {
					who = PSE
					casus_belli = cb_disloyal_vassal
				}
				hidden_effect = {
					PSE = {
						country_event = {
							id = persia.12
						}
					}
				}
			}
		}

		if = {
			limit = {
				exists = ANH
			}
			add_casus_belli = {
				target = ANH
				type = cb_disloyal_vassal
				months = 120
			}
			if = {
				limit = { ai = yes }
				declare_war_with_cb = {
					who = ANH
					casus_belli = cb_disloyal_vassal
				}
				hidden_effect = {
					ANH = {
						country_event = {
							id = persia.12
						}
					}
				}
			}
		}
	}
	# dont Attack Persia
	option = {
		name = "persia.10.b"
		add_prestige = -15
		set_country_flag = attack_on_pse_flag
	}
}

		}

			factor = 0.75
