########################################################################################
# attack the egyptian army  1/2
########################################################################################
country_event = {
	id = judah.1
	title = "judah.1.t"
	desc = "judah.1.d"
	picture = rivercrossing
	fire_only_once = yes

	trigger = {
			tag = JUD
			NOT = { has_country_flag = egyptian_army_passed_throught }
			NOT = {  started_in = 146.1.1 }
			NOT = { is_year = 150 }
			NOT = { alliance_with = EGY }
			NOT = {
				has_opinion = {
					who = EGY
					value = 50
				}
			}
			NOT = {
				any_enemy_country = {
					war_with = EGY
				}
			}
			any_owned_province = {
				OR = {
					units_in_province = EGY
					units_in_province = PHI
				}
			}
	}

	option = {
		name = "judah.1.a"
		ai_chance = { factor = 99  }
		set_country_flag = egyptian_army_passed_throught
		add_prestige = 10
		tooltip = {
			EGY = {
				declare_war_with_cb = {
					who = JUD
					casus_belli = cb_vassalize_mission
				}
			}
		}
		hidden_effect = {
			EGY = {
				country_event = {
					id = judah.2
				}
			}
		}
	}
	option = {
		name = "judah.1.b"
		ai_chance = { factor = 1  }
		add_prestige = -15
		set_country_flag = egyptian_army_passed_throught
	}
}

########################################################################################
# Make Judah or Israel jewish
########################################################################################
country_event = {
	id = judah.4
	title = "judah.4.t"
	desc = "judah.4.d"
	picture = temple

	trigger = {
		OR = {
			tag = JUD
			tag = ISR
		}
		primary_culture = hebrew
		NOT = { religion = jewish }
		NOT = { has_country_flag = jewish_samaritan_conversion_event_once_flag }
	}

	mean_time_to_happen = {
		months = 1
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = { ai = no }
				set_country_flag = jewish_samaritan_conversion_event_once_flag
			}
		}
	}

	option = {
		name = "judah.4.a"
		ai_chance = { factor = 99 }
		change_religion = jewish
		every_owned_province = {
			limit = { culture = hebrew }
			change_religion = jewish
		}
		add_stability = 1
		add_legitimacy = 50
		add_republican_tradition = 50
		add_devotion = 50
	}

	option = {
		name = "judah.4.b"
		ai_chance = { factor = 1 }
		add_stability = -1
		add_legitimacy = -50
		add_republican_tradition = -50
		add_devotion = -50
	}
}
