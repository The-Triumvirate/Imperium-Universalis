# Term Election
country_event = {
	id = elections.1
	title = "elections.1.t"
	desc = "elections.1.d"
	picture = senat
	
	is_triggered_only = yes
	
	trigger = {
		government = republic
		NOT = { has_reform = greek_polis_reform }
		NOT = { has_reform = indian_republic_reform }
		NOT = { has_reform = martial_republic_reform }
		NOT = { has_reform = veche_republic_reform }
		NOT = { has_reform = heaven_republic_reform }
		NOT = { has_reform = pirate_republic_reform }
		NOT = { has_reform = res_publica_reform }
		NOT = { has_reform = late_res_publica_reform }
		if = {
			limit = { has_reform = jin_confederation_reform }
			NOT = { any_owned_province = { region = jinhan_region } }
			NOT = { any_owned_province = { region = byeonhan_region } }
			NOT = { any_owned_province = { region = mahan_region } }
		}
	}

	option = {
		name = "elections.a"	# An administrator type
		ai_chance = { factor = 30 }
		define_ruler = {
			adm = 3
			max_random_dip = 2
			max_random_mil = 2
			claim = 100
			age = 40
			male = yes
		}
		add_adm_power = 20
		add_dip_power = -10
		add_mil_power = -10
	}
	
	option = {
		name = "elections.b"	# A diplomat
		ai_chance = { factor = 30 }
		define_ruler = {
			max_random_adm = 2
			dip = 3
			max_random_mil = 2
			min_age = 25
			max_age = 50
		}
		add_adm_power = -10
		add_dip_power = 20
		add_mil_power = -10
	}
	
	option = {
		name = "elections.c"	# A military man
		ai_chance = { factor = 30 }
		define_ruler = {
			max_random_adm = 2
			max_random_dip = 2
			mil = 3
			min_age = 25
			max_age = 50
		}
		add_adm_power = -10
		add_dip_power = -10
		add_mil_power = 20
	}
	
	option = {
		name = "elections.e"	# the old one
		trigger = {
			republican_tradition = 60
			OR = {
			    has_reform = empower_senatorial_faction_reform
				has_reform = oligarchy_republic_reform
				has_reform = xenophon_republic_reform
		    }
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				NOT = { republican_tradition = 90 }
			}
			modifier = {
				factor = 4
				republican_tradition = 100
			}
		}
		if = {
			limit = { NOT = { has_country_flag = mytilene_pittacus_the_great_flag } }
			add_scaled_republican_tradition = -10
		}
		change_adm = 1
		change_dip = 1
		change_mil = 1
	}
}
