	}

    #force them to disperse (Kingdom)
	option = {
	    name = "slavic.5.a"
		2025 = {
		    spawn_rebels = {
                type = republican_rebels

			}

	}

	    name = "slavic.5.b"

	    name = "slavic.5.c"
