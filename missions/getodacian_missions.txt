	dac1_mission = {
		#position = 1
		icon = dac_army
		required_missions = {  }

			OR = {
				AND = {
					army_professionalism = 0.05
					army_size = 10

						shock = 3
					}
				}

					calc_true_if = {

						}

					}
				}
			}
		}
		effect = {

				limit = {
					NOT = { is_core = ROOT }
