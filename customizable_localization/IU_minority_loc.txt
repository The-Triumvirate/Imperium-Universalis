	name = GetMajorityCultureSize

		localisation_key = AN_supermajority_culture_tooltip

			is_variable_equal = { which = AN_minority_culture_size value = 1 }
			is_variable_equal = { which = AN_secondary_culture value = 0 }
		}
	}
	text = {
		localisation_key = AN_majority_culture_tooltip
		trigger = {
			OR = {
				AND = {

				}
			}
		}
	}

		localisation_key = AN_slim_majority_culture_tooltip
		trigger = {
			OR = {

					check_variable = { which = AN_secondary_culture value = 1 }
				}
			}
		}

		trigger = {
