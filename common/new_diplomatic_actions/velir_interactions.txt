
establish_velir_diplomatic = {
	category = influence
	alert_index = 30
	alert_tooltip = establish_velir_diplomatic_alert_tooltip.1
	
	require_acceptance = yes
	is_visible = {
		is_subject = no
		OR = {
			has_reform = tamil_crowned_king_tribal_reform
			has_reform = tamil_crowned_king_reform
		}
		has_country_flag = can_diplomatically_establish_velir_flag
		FROM = {
			capital_scope = {
				OR = {
					superregion = uttarapatha_superregion
					superregion = dakshinapatha_superregion
				}
			}
			is_subject = no
		}
	}
	is_allowed = {
		hidden_trigger = {
			is_subject = no
			OR = {
				has_reform = tamil_crowned_king_tribal_reform
				has_reform = tamil_crowned_king_reform
			}
			has_country_flag = can_diplomatically_establish_velir_flag
			FROM = {
				capital_scope = {
					OR = {
						superregion = uttarapatha_superregion
						superregion = dakshinapatha_superregion
					}
				}
				is_subject = no
			}
		}
		reverse_has_opinion = {
			who = FROM
			value = 180
		}
		FROM = {
			is_at_war = no
			NOT = { war_with = ROOT }
		}
		NOT = { war_with = FROM }
	}
	on_accept = {
		add_prestige = 4
		create_subject = {
			subject_type = velir_subject
			subject = FROM
		}
	}
	
	ai_acceptance = {
		add_entry = {
			name = TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			multiply_variable = { which = ai_value value = -3 }
		}
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}
			subtract_variable = { which = ai_value value = 70 }
		}
		add_entry = {
			name = ARMY_SIZE
			export_to_variable = {
				variable_name = ai_value
				value = land_forcelimit
				who = ROOT
			}
			change_variable = { which = ai_value value = 1 }
		}
		add_entry = {
			name = STABILITY
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = ROOT
			}
			multiply_variable = { which = ai_value value = 4 }
		}
		add_entry = {
			name = PRESTIGE
			export_to_variable = {
				variable_name = ai_value
				value = prestige
				who = ROOT
			}
			divide_variable = { which = ai_value value = 8 }
		}
		add_entry = {
			name = DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = modifier:diplomatic_reputation
				who = ROOT
			}
			multiply_variable = { which = ai_value value = 7 }
		}
		add_entry = {
			name = DIFFERENT_RELIGION
			limit = { NOT = { religion = FROM } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = BASE_VALUE_CUSTOM
			limit = { FROM = { ai = yes } }
			export_to_variable = {
				which = ai_value
				value = -20
			}
		}
	}
	
	ai_will_do = {
		has_any_disaster = no
		FROM = {
			is_neighbor_of = ROOT
			NOT = { army_size = ROOT }
			NOT = { total_development = ROOT }
		}
		reverse_has_opinion = {
			who = FROM
			value = 160
		}
		stability = 1
		prestige = 10
		NOT = { any_subject_country = { liberty_desire = 50 } }
		is_at_war = no
		NOT = { average_unrest = 1 }
	}
}

part_ways_velir = {
	category = influence
	alert_index = 28
	alert_tooltip = part_ways_velir_alert_tooltip.1
	
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = velir_subject
		}
	}
	is_allowed = {
		always = yes
	}
	on_accept = {
		add_prestige = -25
		add_splendor = -10
		FROM = {
			grant_independence = yes
		}
	}
	
	ai_will_do = {
		always = no
	}
}

centralize_velir_action = {
	category = influence
	alert_index = 6
	alert_tooltip = centralize_velir_action_alert_tooltip.1
	
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = velir_subject
		}
	}
	is_allowed = {
		hidden_trigger = {
			FROM = {
				is_subject_of = ROOT
				is_subject_of_type = velir_subject
			}
		}
		dip_power = 200
		FROM = {
			has_opinion = {
				who = ROOT
				value = 160
			}
			NOT = { liberty_desire = 50 }
		}
	}
	on_accept = {
		add_absolutism = 1
		add_dip_power = -200
		create_subject = {
			subject_type = vassal
    		subject = FROM
		}
		FROM = {
			add_liberty_desire = 10
			add_opinion = {
				who = ROOT
				modifier = decreased_autonomy_opinion
			}
		}
	}
	
	ai_will_do = {
		always = no
	}
}
