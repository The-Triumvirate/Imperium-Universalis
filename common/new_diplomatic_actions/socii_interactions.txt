
#Ask for Socii
ask_for_socii_subject = {
	category = influence
	alert_index = 30
	alert_tooltip = ask_for_socii_subject_alert_tooltip.1

	require_acceptance = yes
	is_visible = {
		is_subject = no
		OR = {
			has_reform = res_publica_reform
			has_reform = roman_kingdom_reform
		}
		NOT = { has_country_flag = no_more_socii }
		government_rank = 2
		FROM = {
			capital_scope = {
				province_group = pg_italica_suburbica_group
			}
			is_subject = no
		}
	}
	is_allowed = {
		hidden_trigger = {
			is_subject = no
			OR = {
				has_reform = res_publica_reform
				has_reform = roman_kingdom_reform
			}
			government_rank = 2
			FROM = {
				capital_scope = {
					province_group = pg_italica_suburbica_group
				}
				is_subject = no
			}
		}
		alliance_with = FROM
		reverse_has_opinion = {
			who = FROM
			value = 150
		}
		FROM = {
			is_at_war = no
			NOT = { war_with = ROOT }
			NOT = { has_country_modifier = recently_removed_socii }
		}
		is_bankrupt = no
		NOT = { war_with = FROM }
	}
	on_accept = {
		add_prestige = 2
		create_subject = {
			subject_type = socii_subject
			subject = FROM
		}
		add_government_power = { mechanic_type = social_dissatisfacton_mechanic power_type = social_dissatisfacton value = 2 }
	}

	ai_acceptance = {
		add_entry = {
			name = TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			multiply_variable = { which = ai_value value = -2 }
		}
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}
			subtract_variable = { which = ai_value value = 70 }
		}
		add_entry = {
			name = ARMY_SIZE
			export_to_variable = {
				variable_name = ai_value
				value = land_forcelimit
				who = ROOT
			}
			change_variable = { which = ai_value value = 1 }
		}
		add_entry = {
			name = STABILITY
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = ROOT
			}
			multiply_variable = { which = ai_value value = 5 }
		}
		add_entry = {
			name = PRESTIGE
			export_to_variable = {
				variable_name = ai_value
				value = prestige
				who = ROOT
			}
			divide_variable = { which = ai_value value = 10 }
		}
		add_entry = {
			name = DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = modifier:diplomatic_reputation
				who = ROOT
			}
			multiply_variable = { which = ai_value value = 6 }
		}
		add_entry = {
			name = LATIN_WAR_WON_LATIN
			limit = { has_country_flag = latin_war_won_flag FROM = { culture = latin } }
			export_to_variable = {
				which = ai_value
				value = 10
			}
		}
		add_entry = {
			name = LATIN_WAR_WON
			limit = { has_country_flag = latin_war_won_flag NOT = { FROM = { culture = latin } } }
			export_to_variable = {
				which = ai_value
				value = 5
			}
		}
		add_entry = {
			name = SOCII_ESTATE_LOYALTY_BAD
			limit = { NOT = { estate_loyalty = { estate = estate_socii loyalty = 50 } } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = SOCII_ESTATE_LOYALTY_NEUTRAL
			limit = { estate_loyalty = { estate = estate_socii loyalty = 50 } NOT = { estate_loyalty = { estate = estate_socii loyalty = 60 } } }
			export_to_variable = {
				which = ai_value
				value = -5
			}
		}
		add_entry = {
			name = SOCII_ESTATE_LOYALTY_HIGH
			limit = { estate_loyalty = { estate = estate_socii loyalty = 60 } }
			export_to_variable = {
				which = ai_value
				value = 5
			}
		}
		add_entry = {
			name = DIFFERENT_RELIGION
			limit = { NOT = { religion = FROM } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = DIFFERENT_CULTURE_GROUP
			limit = { NOT = { culture_group = FROM } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = BASE_VALUE_CUSTOM
			limit = { FROM = { ai = yes } }
			export_to_variable = {
				which = ai_value
				value = -20
			}
		}
	}

	ai_will_do = {
		has_any_disaster = no
		FROM = {
			is_neighbor_of = ROOT
			NOT = { army_size = ROOT }
			NOT = { total_development = ROOT }
		}
		reverse_has_opinion = {
			who = FROM
			value = 150
		}
		stability = 1
		army_size = 10
		is_at_war = no
		NOT = { average_unrest = 1 }
	}
}

#Frees Socii
remove_socii_subject = {
	category = influence
	alert_index = 41
	alert_tooltip = remove_socii_subject_alert_tooltip.1

	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of_type = socii_subject
			is_subject_of = ROOT
		}
	}
	is_allowed = {
		hidden_trigger = {
			FROM = {
				is_subject_of_type = socii_subject
				is_subject_of = ROOT
			}
		}
		NOT = {
			has_country_modifier = 10_percent_over_ratio_rome
			has_country_modifier = 25_percent_over_ratio_rome
			has_country_modifier = 50_percent_over_ratio_rome
		}
		is_at_war = no
		has_any_disaster = no
		FROM = {
			is_subject_of_type = socii_subject
			is_subject_of = ROOT
		}
	}
	on_accept = {
		FROM = {
			grant_independence = yes
			add_truce_with = ROOT
			add_country_modifier = {
				name = recently_removed_socii
				duration = 7300
			}
			add_opinion = {
				who = ROOT
				modifier = break_alliance
			}
		}
		add_opinion = {
			who = FROM
			modifier = break_alliance
		}
		add_government_power = { mechanic_type = social_dissatisfacton_mechanic power_type = social_dissatisfacton value = -2 }
		add_prestige = -5
		add_mil_power = -20
	}

	ai_will_do = {
		FROM = {
			OR = {
				total_development = ROOT
				liberty_desire = 80
			}
		}
	}
}

# Call Socii into wars
call_socii_into_wars = {
	category = influence

	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = socii_subject
		}
	}
	is_allowed = {
		if = {
			limit = {
				FROM = { liberty_desire = 50 }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_REBELLIOUS_TT
				FROM = { liberty_desire = 50 }
			}
		}
		if = {
			limit = {
				is_at_war = no
			}
			custom_trigger_tooltip = {
				tooltip = ROOT_IS_NOT_AT_WAR_TT
				is_at_war = yes
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = {
						owes_favors = {
							who = ROOT
							value = 5
						}
					}
				}
				is_in_war = {
					defender_leader = ROOT
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_MUST_OWE_5_FAVORS_DEFENSIVE_WAR_TT
				FROM = {
					owes_favors = {
						who = ROOT
						value = 5
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = {
						owes_favors = {
							who = ROOT
							value = 10
						}
					}
				}
				NOT = {
					is_in_war = {
						defender_leader = ROOT
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_MUST_OWE_10_FAVORS_TT
				FROM = {
					owes_favors = {
						who = ROOT
						value = 10
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { is_at_war = yes }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_AT_WAR_TT
				FROM = { is_at_war = no }
			}
		}
	}
	on_accept = {
		if = {
			limit = {
				is_in_war = {
					defender_leader = ROOT
				}
			}
			FROM = {
				ROOT = {
					add_favors = {
						who = PREV
						amount = -5
					}
				}
				join_all_defensive_wars_of = ROOT
			}
		}
		if = {
			limit = {
				ROOT = {
					is_in_war = {
						attacker_leader = ROOT
					}
				}
				FROM = {
					owes_favors = {
						who = ROOT
						value = 10
					}
				}
			}
			FROM = {
				ROOT = {
					add_government_power = { mechanic_type = social_dissatisfacton_mechanic power_type = social_dissatisfacton value = 0.25 }
					add_favors = {
						who = PREV
						amount = -10
					}
				}
				add_liberty_desire = 5
				join_all_offensive_wars_of = ROOT
				join_all_defensive_wars_of = ROOT
			}
		}
	}
	on_decline = {
	}

	ai_will_do = { # this is a trigger
		always = yes
	}
}
