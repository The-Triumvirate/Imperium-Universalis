							owner = { has_reform = free_city }

							owner = { is_elector = yes }
							development = 20
						}
						development = 30
						province_id = 4349 # Regensburg
						#province_id = 1868 # Augsburg
						#province_id = 1872 # Ulm
					}
				}
				province_event = {
					id = hre_event.6
				}
				add_province_triggered_modifier = perpetual_diet
			}
		}
	}
	off_effect = {
