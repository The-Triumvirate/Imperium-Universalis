######################################################################
# Nationalist rebels. Freedom!!!
######################################################################
nationalist_rebels = {

	color = { 25 180 25 }

	area = nation_rebel_tag 			# Stays within provinces of their rebel tag
	government = any				# No special preference
	defection = nation_rebel_tag			# If the country seeks independence, defection specifies what type of provinces it wants.
	independence = nation_rebel_tag 		# Seeks independence for spawn province culture. Will not spawn at all if there is no suitable tag.
	defect_delay = 36				# Can spontaneously declare independence after 18 months (earliest)
	unit_transfer = yes				# Units switch allegiance after enforcing demands.
	gfx_type = culture_province
	will_relocate = yes

	resilient = no
	reinforcing = yes
	general = yes
	smart = yes

	artillery = 0.3
	infantry = 0.6
	cavalry = 0.1

	morale = 1.0

	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes
	handle_action_change_culture = yes

	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 2

		modifier = {
			factor = 5
			owner = { is_tribal = yes }
		}
		modifier = {
			factor = 5
			is_territory = yes
		}
		modifier = {
			factor = 5
			nationalism = 1
		}
		modifier = {
			factor = 0.001
			is_colony = yes
		}
		modifier = {
			factor = 0.75
			is_overseas = yes
			has_owner_culture = yes
		}
		modifier = {
			factor = 0.8
			owner = {
				stability = 1
			}
		}
		modifier = {
			factor = 0.8
			owner = {
				stability = 2
			}
		}
		modifier = {
			factor = 0.8
			owner = {
				stability = 3
			}
		}
		modifier = {
			factor = 2
			NOT = { is_core = owner }
		}
		modifier = {
			factor = 3
			owner = {
				NOT = { stability = 0 }
			}
		}
		modifier = {
			factor = 3
			owner = {
				NOT = { stability = -2 }
			}
		}
		modifier = {
			factor = 2.5
			has_owner_culture = no
		}
		modifier = {
			factor = 5
			owner = { NOT = { culture_group = ROOT } }
		}
		modifier = {
			factor = 1.5
			NOT = { tolerance_to_this = 1 }
		}
		modifier = {
			factor = 1.5
			NOT = { tolerance_to_this = -1 }
		}
		modifier = {
			factor = 1.5
			NOT = { tolerance_to_this = -2 }
		}
		modifier = {
			factor = 2
			owner = {
				ai = yes
				luck = no
			}
		}
		modifier = {
			factor = 5
			is_in_capital_area = no
			is_island = no
		}
		modifier = {
			factor = 10
			owner = {
				NOT = { num_of_ports = 1 }
			}
			is_in_capital_area = no
		}
		modifier = {
			factor = 2.0
			owner = {
				war_exhaustion = 4
			}
		}
		modifier = {
			factor = 2.0
			owner = {
				war_exhaustion = 8
			}
		}
	}

	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}

	# Province Scope
	siege_won_trigger = {
		culture = REB
	}
	siege_won_effect = {
		add_nationalism = 10 # Add ten extra years of nationalism
		if = {
			limit = { has_province_modifier = building_colony }
			change_variable = { which = colony_progress value = -20 }
		}
	}

	# Country scope
	can_negotiate_trigger = {
		is_at_war = no
	}

	# Country scope
	can_enforce_trigger = {
		always = yes
	}

	# Localisation for their demands
	demands_description = "nationalist_rebels_demand"

	# Country scope
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		add_prestige = -10
		if = {
			limit = {
				exists = REB
			}
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
					culture = REB
				}
				if = {
					limit = { NOT = { is_core = REB } }
					add_core = REB
				}
				cede_province = REB
			}
		}
		if = {
			limit = {
				NOT = { exists = REB }
			}
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
					culture = REB
					NOT = { is_core = REB }
					culture = REB
				}
				add_core = REB
			}
			create_vassal = REB
		}
	}
}
