#Build shrine
estate_merchants_build_market_in_y = {
	max_days_active = 3650
	can_select = {
		dip_tech = 4
		any_owned_province = {
			is_state_core = root
			NOT = { has_building = small_market }
			NOT = { has_building = medium_market }
			NOT = { has_building = great_market }
			num_free_building_slots = 1
		}
	}
	selection_weight = {
		factor = 30
		modifier = {
			factor = 0.75
			NOT = {
				any_owned_province = {
					is_state_core = root
					NOT = { base_production = 1 }
					num_free_building_slots = 1
				}
			}
		}
		modifier = {
			factor = 2.0
			any_owned_province = {
				is_state_core = root
				base_production = 1
				NOT = { has_building = medium_market }
				NOT = { has_building = great_market }
				AND = {
					NOT = { has_building = small_market }
					num_free_building_slots = 1
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				crown_land_share = 15
			}
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { base_production = 1 }
				NOT = { has_building = small_market }
				NOT = { has_building = medium_market }
				NOT = { has_building = great_market }
				num_free_building_slots = 1
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				base_production = 1
				NOT = { has_building = small_market }
				NOT = { has_building = medium_market }
				NOT = { has_building = great_market }
				num_free_building_slots = 1
			}
			save_event_target_as = agenda_province
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
	task_requirements = {
		event_target:agenda_province = {
			OR = {
				has_building = small_market
				has_building = medium_market
				has_building = great_market
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_merchants
			loyalty = 10
		}
		if = {
			limit = {
				NOT = { crown_land_share = 15 }
				estate_territory = {
					estate = estate_merchants
					territory = 5
				}
			}
			take_estate_land_share_small = { estate = estate_merchants }
		}
		else = {
			add_mercantilism = 1
			add_treasury = 25
			add_dip_power = 50
		}
	}
	failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_unrest = 5
			}
		}
		add_estate_loyalty_modifier = {
			estate = estate_merchants
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}

			factor = 0
			ruler_has_personality = conqueror_personality
		}
		modifier = {
			factor = 2
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			war_exhaustion = 4
		}
		modifier = {
			factor = 2
			has_wartaxes = yes
		}
	}

	fail_if = {
		OR = {
			is_at_war = yes

	}

	}
}
