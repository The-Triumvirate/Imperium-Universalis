pay_crown_land = {
    trigger = {
	    crown_land_share = 20
		OR = {
		    has_estate = estate_nobility
			has_estate = estate_priesthood
			has_estate = estate_merchants
		}
	}
	
	effect = {
	    if = {
		    limit = {
			    has_estate = estate_nobility
			}
			change_estate_land_share = {
				estate = estate_nobility
				share = 0.25
			}
		}
		if = {
		    limit = {
			    has_estate = estate_priesthood
			}
			change_estate_land_share = {
				estate = estate_priesthood
				share = 0.25
			}
		}
		if = {
		    limit = {
			    has_estate = estate_merchants
			}
			change_estate_land_share = {
				estate = estate_merchants
				share = 0.25
			}
		}
		else = {
		    add_prestige = -10
		}
	}
	
	chance = {
		factor = 1
	}

	ai_will_do = {
		factor = 0.75
		modifier = {
			factor = 0
			NOT = { crown_land_share = 60 }
		}
	}
}
