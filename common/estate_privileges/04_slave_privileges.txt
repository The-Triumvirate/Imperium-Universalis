###

# 	It will be several topics, grouped in sliders and improvements
#	Most of the beneffits of each law will not be seen in a game modifier, but will be hard-coded for the mod's events.
#	So don't worry if you don't see much modifiers in a given privilege; they will still be taken into account.

# Laws, five topics that will improve their condition. Many of them are required to abolish slavery. Some are negative
# Each will take a bit of the "general income" gotten every five years. Instead of the 40%, it will be down until 20%

#estate_slaves_law_D2

#estate_slaves_law_E2

# Some other notes
# Loyalty means the loyalty of slaves, and will determine risk of general rebellion. There will be events to temporarily affect it
#	It will also affect the disasters put in place, for slave rebellion (Spartacus)

# Influence is that of the slaves over the state. If it reaches high, slaves will try to lower the "bad" privileges  ## removed fow now
#	If influence reaches very high, due to special laws passed, it allows to start the process of Abolishing Slavery ## influence = 0

estate_slaves_status_1 = {		# War captives
	icon = privilege_status_1
	land_share = 0		max_absolutism = 0		loyalty = 0.10	influence = 0
	### Potential and trigger
	is_valid = { # Potential
		always = yes
	}
	can_select = { # Later on it will be fixed, not it's just experimental
		OR = {
			NOT = { has_estate_privilege = estate_slaves_status_1 }
			has_estate_privilege = estate_slaves_status_2
			has_estate_privilege = estate_slaves_status_3
		}
		# Later on, War Captives is the default option for natives and for tribals,
		# when they move into civilized (monarchy/republic/oligarchy/etc) they will automatically change to Serfs
	}
	can_revoke = { always = no }
	### Effects
	on_granted = {
		hidden_effect = {
			remove_estate_privilege = estate_slaves_status_2
			remove_estate_privilege = estate_slaves_status_3
			remove_estate_privilege = estate_slaves_status_4
			remove_estate_privilege = estate_slaves_status_5
			remove_estate_privilege = estate_slaves_status_6
		}
		# stability hit, depending on size of slaves
		if = {		limit = { check_variable = { slave_pop_social = 100 } } add_stability = -3 }
		else_if = { limit = { check_variable = { slave_pop_social = 50 } } 	add_stability = -2 }
		else_if = { limit = { check_variable = { slave_pop_social = 20 } } 	add_stability = -1 }

	}
	on_revoked = {
	}
	benefits = {
		slavesmod_state_income = 0.20	# base value 30%, so, -10%
		slavesmod_liberated = 0.01
		slavesmod_enslav_eff = 0.10
		slaves_privilege_slots = -2	# base 6, reduced to 4, only the three laws
	}

	cooldown_years = 40
	ai_will_do = { factor = 0 }
}
